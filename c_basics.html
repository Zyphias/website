Scanning and Storing Variables


    #include <stdio.h>

    int main(void) {

        int a = 0;
        printf("Enter an integer: ");
        scanf("%i\n", &a);

        printf("The number you entered was %i\n", a);

        return 0;
    }
$ gcc variables.c -o variables $ ./variables $ Enter an integer: 3 $ The number you entered was 3
